function drawWaizi(x, y) { ctx.save(); ctx.translate(x, y); // 小奶猫身体比例：脑袋大，身体小 const headRadius = 14; // 圆脑袋半径 const bodySize = 10; // 身体大小 // 身体（小椭圆，在脑袋下方） ctx.fillStyle = '#c0c0ff'; ctx.beginPath(); ctx.ellipse(0, headRadius * 0.8, bodySize * 0.7, bodySize, 0, 0, Math.PI * 2); ctx.fill(); // 圆脑袋 ctx.beginPath(); ctx.arc(0, 0, headRadius, 0, Math.PI * 2); ctx.fill(); // 软塌塌的小耳朵（用贝塞尔曲线画） ctx.beginPath(); ctx.moveTo(-headRadius * 0.6, -headRadius * 0.5); ctx.quadraticCurveTo(-headRadius * 0.9, -headRadius * 1.4, -headRadius * 0.3, -headRadius * 1.1); ctx.fillStyle = '#c0c0ff'; ctx.fill(); ctx.beginPath(); ctx.moveTo(headRadius * 0.6, -headRadius * 0.5); ctx.quadraticCurveTo(headRadius * 0.9, -headRadius * 1.4, headRadius * 0.3, -headRadius * 1.1); ctx.fill(); // 眼睛（又圆又亮） ctx.fillStyle = '#ffffff'; ctx.beginPath(); ctx.arc(-5, -2, 4, 0, Math.PI * 2); ctx.fill(); ctx.beginPath(); ctx.arc(5, -2, 4, 0, Math.PI * 2); ctx.fill(); ctx.fillStyle = '#0a0a1a'; ctx.beginPath(); ctx.arc(-5, -2, 2.5, 0, Math.PI * 2); ctx.fill(); ctx.beginPath(); ctx.arc(5, -2, 2.5, 0, Math.PI * 2); ctx.fill(); // 小鼻子 ctx.fillStyle = '#ffb0b0'; ctx.beginPath(); ctx.arc(0, 3, 1.8, 0, Math.PI * 2); ctx.fill(); // 短粗的尾巴 ctx.beginPath(); ctx.moveTo(bodySize * 0.6, headRadius * 0.8); ctx.quadraticCurveTo(18, headRadius * 1.2, 16, headRadius * 0.2); ctx.strokeStyle = '#c0c0ff'; ctx.lineWidth = 4; ctx.lineCap = 'round'; ctx.stroke(); // 奶猫的小爪子（前爪） ctx.fillStyle = '#c0c0ff'; ctx.beginPath(); ctx.arc(-6, headRadius * 0.6, 3, 0, Math.PI * 2); ctx.fill(); ctx.beginPath(); ctx.arc(6, headRadius * 0.6, 3, 0, Math.PI * 2); ctx.fill(); // 猫薄荷（小奶猫叼着更显调皮） ctx.beginPath(); ctx.moveTo(8, -8); ctx.lineTo(18, -14); ctx.strokeStyle = '#80ff80'; ctx.lineWidth = 2; ctx.stroke(); ctx.restore(); }
